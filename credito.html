    Criando um DataFrame com as medianas para substituir os valores nulos.¶
    • Na coluna PORTE vou substituir pela MODA.
    • As colunas NO_FUNCIONARIOS, RENDA_MES_ANTERIOR_mediana e NO_FUNCIONARIOS_mediana substituirei pela MÉDIA de cada PORTE
    • Transformação de variaveis numericas em categoricas.
    # Agrupando por ID_CLIENTE e calcula a mediana das colunas RENDA_MES_ANTERIOR e NO_FUNCIONARIOS de cada cliente
mediana_por_cliente = base_info_merged.groupby(['ID_CLIENTE','PORTE']).agg({
    'RENDA_MES_ANTERIOR': 'median',
    'NO_FUNCIONARIOS': 'median'
}).reset_index().sort_values('ID_CLIENTE')

    colunas = ['RENDA_MES_ANTERIOR', 'NO_FUNCIONARIOS']

# preenchendo nulos com a mediana do grupo PORTE
for coluna in colunas:
    medianas = mediana_por_cliente.groupby('PORTE')[coluna].transform('median')
    mediana_por_cliente[coluna] = mediana_por_cliente[coluna].fillna(medianas)
print(mediana_por_cliente.isnull().sum())
